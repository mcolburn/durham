\itId{sys}{project}{roles}{en}{translator}{%
Creates a translation for a given domain (language, country, and realm).
}%
\itId{sys}{project}{roles}{en}{editor}{%
Edits a translation for a given domain (language, country, and realm).
}%
\itId{sys}{project}{roles}{en}{reviewer}{%
Reads through a translation for a given domain (language, country, and realm). Provides feedback regarding the quality of the translation.  Identifies issues with the translation.  Suggests how the translation can be improved.
}%
\itId{sys}{project}{roles}{en}{keyboarder}{%
Enters the text of translation for a given domain (language, country, and realm) into the format required for the \ltOcmcSystemAcronymn.
}%
\itId{sys}{project}{roles}{en}{coordinator}{%
Sets up a \ltOcmcSystemAcronymn, and coordinates with project members to help them use the technologies for the project.
}%
